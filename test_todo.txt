Function to get every files where "ft_" is forgot:

for file in *.c; do
  if ! grep -E "^[^/]*ft_" "$file" | grep -v "static" | grep -v ";" | grep -v "if" > /dev/null; then
    echo "$file"
  fi
done

Function to print every files and the protoypage of the principal function:
grep -H "^[^/]*ft_" *.c | grep -v "static" | grep -v ";" | grep -v "if" | awk -F: '!seen[$1]++ {print $1 "\t\t" $2}'

Function that checks, if there's constant like NULL or size_t without the matching library imported:
for file in *.c; do
  # Check for "NULL" and the corresponding #include <stdlib.h>
  if grep -q "NULL" "$file"; then
    if ! grep -q '#include <stdlib.h>' "$file"; then
      echo "Missing #include <stdlib.h> in $file"
    fi
  fi

  # Check for "size_t" and the corresponding #include <stddef.h>
  if grep -q "size_t" "$file"; then
    if ! grep -q '#include <stddef.h>' "$file"; then
      echo "Missing #include <stddef.h> in $file"
    fi
  fi
done
